from sklearn.base import BaseEstimator, ClusterMixin
from sklearn.neighbors import NearestNeighbors

from sklearn.cluster._dbscan_inner import dbscan_inner

class DBSCAN(BaseEstimator, ClusterMixin):
    
    def __init__(self, eps=0.5, min_samples=5):
        self.eps = eps
        self.min_samples = min_samples
        
    def fit(self, X):
        n_model = NearestNeighbors(
            radius=self.eps, algorithm='auto',
            leaf_size=30, metric='euclidean')
        n_model.fit(X)  #      
        neighborhoods = n_model.radius_neighbors(X,
                                                return_distance=False)  #  

        n_neighbors = np.array([len(neighbors)
                                    for neighbors in neighborhoods])  # -       

        res = np.full(X.shape[0], -1, dtype=np.intp)  #   -1 (   )
 
        core_samples = np.asarray(n_neighbors >= self.min_samples,
                                  dtype=np.uint8)  #  1,    >=  -  (min_samples)
    
        dbscan_inner(core_samples, neighborhoods, res)  #  - (      
                                                        #  n_model.radius_neighbors; ,    < min_samples
                                                        #    ,  -1)
 
        self.res_ = res
        return self
    
    def predict(self):
        return self.res_
    
    def fit_predict(self, x):
        self.fit(x)
        return self.res_

import numpy as np
import matplotlib.pyplot as plt
import time

from sklearn import datasets
from sklearn.preprocessing import StandardScaler

np.random.seed(0)

n_samples = 1500
noisy_circles = datasets.make_circles(n_samples=n_samples, factor=.5,
                                      noise=.05)
noisy_moons = datasets.make_moons(n_samples=n_samples, noise=.05)
blobs = datasets.make_blobs(n_samples=n_samples, random_state=8)
no_structure = np.random.rand(n_samples, 2), None

plt.figure(figsize=(2 + 3, 9.5))

data = [noisy_circles, noisy_moons, blobs, no_structure]
start_time = time.time()
for i_dataset, dataset in enumerate(data):
    X, y = dataset
    # normalize dataset for easier parameter selection
    X = StandardScaler().fit_transform(X)
    
    start_time = time.time()
    dbscan = DBSCAN(eps=0.2, min_samples=3).fit(X)
    labels = dbscan.predict()

    # plot
    plt.subplot(4, 1, i_dataset + 1)
    plt.scatter(X[:, 0], X[:, 1], c=labels, s=10)
    plt.xlim(-2, 2)
    plt.ylim(-2, 2)
    plt.xticks(())
    plt.yticks(())
print('My realization takes %s sec' % (time.time() - start_time))

plt.show()
